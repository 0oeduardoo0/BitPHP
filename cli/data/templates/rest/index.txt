<?php
	/**
	 *	RESTful app index template
	 */

	//Initialize rest service
	require('core/bit.php');
	$app = $bitphp->loadApiServer();

	// Create a get route to '/api/v1/hello'
	$app->get('/api/v1/hello', function() use ($app) {
		$app->response([
			  'status' => $app->getStatusMessage()
			, 'result' => 'Hola mundo!'
		]);
	});

	$app->start();
